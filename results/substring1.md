int substring1(int* a) {
	int j = a[2];
	j = 0;
	int i = a[0];
	int k = a[1];

	iif_assume ((i >= 0) && (i <= k) && (k >= 0) && (k <= 100)); 
	while (i < k) {
		recordi(i, k, j);
		i++;	
		j++;
	}
	recordi(i, k, j);
	iif_assert(j < 101);
	return 0;
}
