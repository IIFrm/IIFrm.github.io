## program
```c
int func(int* a)
{
    int x = 0;
    int y = a[0];
    iif_assume(y >= 0);
    while (x < 99) {
        recordi(y);
        if (y % 2 == 0)
            x++;
        else 
            x += 2;
    }
    recordi(y);
    iif_assert((x % 2) != (y % 2));
    return 0;
}
```
