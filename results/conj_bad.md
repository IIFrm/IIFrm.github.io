int conj(int* a)
{
	int x;
	x = a[0];
	int loop_times = a[1] % 10;
	iif_assume((x >= 0) && (x <= 50));
	while (loop_times-- > 0) {
		recordi(x, loop_times);
		if (x >= 50) {
			x --;
		}
		else if (x <= 0) {
			x ++;
		}
		else {
			x += rand() %3 - 1;
		}
	}
	recordi(x, loop_times);
	iif_assert((x >= 0) && (x <= 50));
	return 0;
}
